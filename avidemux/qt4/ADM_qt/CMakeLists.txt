SET(ADM_LIB ADM_qtx_${QT_EXTENSION})

ADD_LIBRARY(${ADM_LIB} STATIC)
TARGET_SOURCES(${ADM_LIB} PRIVATE src/ADM_qtx.cpp)

TARGET_INCLUDE_DIRECTORIES(${ADM_LIB} PUBLIC ${CMAKE_CURRENT_BINARY_DIR} ${QT_INCLUDE_DIR})
TARGET_INCLUDE_DIRECTORIES(${ADM_LIB} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)
SET_TARGET_PROPERTIES(${ADM_LIB} PROPERTIES CXX_VISIBILITY_PRESET hidden)
SET_TARGET_PROPERTIES(${ADM_LIB} PROPERTIES VISIBLITY_INLINES_HIDDEN True)
TARGET_LINK_LIBRARIES(${ADM_LIB} PUBLIC
    ADM_core6
    ${QT_QTGUI_LIBRARY}
    ${QT_QTCORE_LIBRARY}
)
